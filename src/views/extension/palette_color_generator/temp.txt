<script setup>
import { ref, reactive, onMounted } from 'vue';
import { generateCustomPalette, isLight, generateCssVariables } from "../../../utils/colorHelper";

const hexInput = ref("#02c69a");
const palette = ref([]);
const copied = ref("");

// Cấu hình mặc định
const config = reactive({
  step: 0.1, // Khoảng cách độ sáng
  saturationScale: 1.0,
  count: 9
});

const sliderDefinitions = {
  step: { label: "Lightness Interval", min: 0.05, max: 0.2, step: 0.01, unit: "" },
  saturationScale: { label: "Saturation Scale", min: 0.1, max: 2, step: 0.1, unit: "x" },
  count: { label: "Color Steps", min: 5, max: 11, step: 2, unit: " swatches" }
};

// Thiết lập giá trị mặc định chuẩn để tái sử dụng
const DEFAULT_CONFIG = {
  step: 0.1,
  saturationScale: 1.0,
  count: 9
};

const resetConfig = () => {
  // Reset từng giá trị trong reactive object
  config.step = DEFAULT_CONFIG.step;
  config.saturationScale = DEFAULT_CONFIG.saturationScale;
  config.count = DEFAULT_CONFIG.count;

  // Cập nhật lại dải màu ngay lập tức
  update();
};

// // Hàm cập nhật linh hoạt cho Slider
// const update = () => {
//   let val = hexInput.value.trim();
//   if (!val.startsWith('#')) val = '#' + val;

//   if (/^#([A-Fa-f0-9]{3}){1,2}$/.test(val)) {
//     palette.value = generateCustomPalette(val, config);
//   }
// };
const update = () => {
  let val = hexInput.value.trim();

  if (val) {
    // Logic "Sạch": Xóa toàn bộ dấu # hiện có, sau đó thêm lại đúng 1 dấu ở đầu
    val = '#' + val.replace(/#/g, '');
    hexInput.value = val;
  }

  // Regex kiểm tra mã màu Hex chuẩn (3 hoặc 6 ký tự)
  if (/^#([A-Fa-f0-9]{3}){1,2}$/.test(val)) {
    palette.value = generateCustomPalette(val, config);
  }
};

// Hàm cập nhật "Cứng" khi nhấn ENTER hoặc nút GENERATE
const handleHardUpdate = () => {
  // Khi người dùng chủ động nhấn Generate/Enter, ta reset về thông số "vàng" 
  // để màu nhập vào hiển thị đẹp nhất ngay lập tức.
  config.step = 0.1;
  config.saturationScale = 1.0;
  update();
};

// const copy = (color) => {
//   if (!navigator.clipboard) return;
//   navigator.clipboard.writeText(color);
//   copied.value = color;
//   setTimeout(() => (copied.value = ""), 2000);
// };
const copy = async (color) => {
  try {
    await navigator.clipboard.writeText(color);
    copied.value = color;
    setTimeout(() => (copied.value = ""), 2000);
  } catch {
    alert("Không copy được (có thể do HTTP)");
  }
};

const exportCSS = () => {
  const css = generateCssVariables(palette.value);
  navigator.clipboard.writeText(css);
  // Bạn có thể thay alert bằng một toast đẹp hơn nếu muốn
  copied.value = "CSS Copied!";
  setTimeout(() => (copied.value = ""), 2000);
};

onMounted(update);

const pasteColor = async () => {
  try {
    // Đọc văn bản từ bộ nhớ tạm
    const text = await navigator.clipboard.readText();

    // Kiểm tra xem có phải mã màu HEX hợp lệ không (ví dụ: #ffffff hoặc ffffff)
    const hexRegex = /^#?([0-9A-F]{3}){1,2}$/i;

    if (hexRegex.test(text.trim())) {
      // Nếu không có dấu # thì thêm vào cho chuẩn
      const validHex = text.startsWith('#') ? text.trim() : `#${text.trim()}`;

      hexInput.value = validHex;

      // Gọi hàm update để tạo lại dải màu ngay lập tức
      if (typeof update === 'function') {
        update();
      }

      console.log("Pasted & Updated:", validHex);
    } else {
      alert("Nội dung trong Clipboard không phải mã màu HEX hợp lệ!");
    }
  } catch (err) {
    console.error('Không thể truy cập Clipboard:', err);
    alert("Vui lòng cho phép trình duyệt truy cập Clipboard.");
  }
};

</script>
